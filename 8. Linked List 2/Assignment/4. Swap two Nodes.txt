Node *swapNodes(Node *head, int i, int j)
{
	//Write your code here
    if(head == NULL || i == j){
        return head;
    }
    Node *iprev=head, *jprev=head;
    for(int count=0; count<i-1; count++){
        iprev=iprev->next;
        if(iprev == NULL){
            return head;
        }
    }
    
    for(int count=0; count<j-1; count++){
        jprev= jprev->next;
        if(jprev == NULL){
            return head;
        }
    }
    
    
    if(iprev->next == jprev){
    	Node *temp= iprev->next;
        iprev->next= jprev->next;
        Node *temp2= jprev->next;
        jprev->next= jprev->next->next;
        temp2->next= temp;
        return head;
    }
    
    Node *temp= iprev->next->next;
    iprev->next->next= jprev->next->next;
	jprev->next->next= temp;
    temp= iprev->next;				//ith element
    iprev->next= jprev->next;		// now iprev is pointing to the jth element
    jprev->next= temp;					//now jprev is pointing to the ith element
    return head;
}
